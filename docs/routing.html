int intParser(String value) #

Parses an integer, returning the integer on success, or null on failure.

int intParser(String value) {
 try {
   return int.parse(value);
 } on FormatException {
   return null;
 }
}

List<Tuple> zip(List fsts, List snds) #

List<Tuple> zip(List fsts, List snds) {
 var z = <Tuple>[];
 int l = math.min(fsts.length, snds.length);
 for (int i = 0; i < l; i++) {
   z.add(new Tuple(fsts[i], snds[i]));
 }
 return z;
}
